    HYPER.Physics.Material = function (e) {
        e = e || {};

        this.dynamicFriction = e.dynamicFriction || e.friction || 0;
        /**
          friction of object when still.
         */
        this.staticFriction = e.staticFriction || e.friction || 0;
        /**
          How bouncy an object is.
         */
        this.bounciness = e.bounciness || 0;
        /**
          What other Physics Matearials this one will interact with.
         */
        this.interact = e.interact || "all";
        /**
          What other Physics Matearials this one will exclude from interacting with.
         */
        this.exclude = e.exclude || "none";
        return this;
